// @lc code=start
func addBinary(a string, b string) string {
	var carry, sum int
	i, j := len(a), len(b)
	if i < j {
		i, j = j, i
		a, b = b, a
	}

	res := make([]byte, i+1)

	for j > 0 {
		j--
		i--
		sum = int(a[i]-'0') + int(b[j]-'0') + carry
		carry = sum / 2
		sum = sum % 2
		res[i+1] = byte(sum + '0')
	}

	for i > 0 {
		i--
		sum = int(a[i]-'0') + carry
		carry = sum / 2
		sum = sum % 2
		res[i+1] = byte(sum + '0')
	}

	res[0] = byte(carry + '0')

	for i < len(res)-1 {
		if res[i] == '0' {
			i++
		} else {
			break
		}
	}
	return string(res[i:])
}
