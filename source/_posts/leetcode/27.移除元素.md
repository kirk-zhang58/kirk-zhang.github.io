// @lc code=start
func removeElement(nums []int, val int) int {
	slow, fast := 0, 0
	for fast < len(nums) {
		if val != nums[fast] {
			nums[slow] = nums[fast]
			slow++
		}
		fast++

	}
	return len(nums[:slow])
}
