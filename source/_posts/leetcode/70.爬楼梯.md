func climbStairs(n int) int {

	if n == 0 {
		return 1
	}
	if n == 1 {
		return 1
	}
	if n == 2 {
		return 2
	}
	dp := []int{1, 2}
	for i := 3; i <= n; i++ {
		Sum(dp)
	}

	return dp[1]
}
func Sum(dp []int) {
	pre := dp[0]
	dp[0] = dp[1]
	dp[1] = pre + dp[1]
}
